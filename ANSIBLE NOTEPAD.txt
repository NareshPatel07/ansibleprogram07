Playbook.yml File

- name: Configure Web Server
  hosts: webservers
  become: true
  tasks:
    - name: Install Apache
      yum:
        name: httpd
        state: latest

    - name: Start and enable Apache
      service:
        name: httpd
        state: started
        enabled: true

    - name: Create a custom index.html
      copy:
        dest: /var/www/html/index.html
        content: "<h1>Hello from Ansible Web Server!</h1>"

- name: Configure DB Server
  hosts: dbservers
  become: true
  tasks:
    - name: Install Apache
      yum:
        name: httpd
        state: latest

    - name: Start and enable Apache
      service:
        name: httpd
        state: started
        enabled: true
